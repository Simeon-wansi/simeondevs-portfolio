```javascript
// Load featured projects on homepage
async function loadFeaturedProjects() {
    const { data, error } = await supabaseClient
        .from('projects')
        .select('*')
        .eq('published', true)
        .eq('featured', true)
        .order('created_at', { ascending: false })
        .limit(3);
    
    if (error) {
        console.error('Error loading featured projects:', error);
        return;
    }
    
    const container = document.getElementById('featuredProjectsGrid');
    if (!container) return;
    
    container.innerHTML = '';
    data.forEach(project => {
        const card = createProjectCard(project);
        container.appendChild(card);
    });
}

// Call on page load
window.loadFeaturedProjects = loadFeaturedProjects;
```

**In `js/main.js`**, add to DOMContentLoaded:

```javascript
// Load latest blog posts for homepage
async function loadLatestBlogPosts() {
    const { data, error } = await supabaseClient
        .from('blog_posts')
        .select('*')
        .eq('status', 'published')
        .order('published_at', { ascending: false })
        .limit(3);
    
    if (error) {
        console.error('Error loading latest blog posts:', error);
        return;
    }
    
    const container = document.getElementById('latestBlogGrid');
    if (!container) return;
    
    container.innerHTML = '';
    data.forEach(post => {
        const card = createBlogCard(post);
        container.appendChild(card);
    });
}

// Export
window.loadLatestBlogPosts = loadLatestBlogPosts;
```

**In `js/main.js`**, add to DOMContentLoaded (after featured projects):
```javascript
// Load latest blog posts on homepage
if (typeof loadLatestBlogPosts === 'function') {
    loadLatestBlogPosts();
}
```
